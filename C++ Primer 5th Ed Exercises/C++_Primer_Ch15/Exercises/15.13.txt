class base {
public:
string name() { return basename; }
virtual void print(ostream &os) { os << basename; }
private:
string basename;
};
class derived : public base {
public:
//no base scope operator infinite call to itself, base::print(os)
//calls print which returns basename
void print(ostream &os) { print(os); os << " " << i; }
private:
int i;
};
